SELECT I.ID, I.SUMMARY, I.ISSUESLIST_ID, IL.NAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID
ORDER BY I.ISSUESLIST_ID, I.ID;

SELECT I.ID, I.SUMMARY, I.USER_ID_ASSIGNEDTO, U.ID, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY I.ISSUESLIST_ID, I.ID;

SELECT U.ID, U.FIRSTNAME, U.LASTNAME,
SUM(I.USER_ID_ASSIGNEDTO = U.ID) AS ISSUES_QUANTITY
FROM ISSUES I, USERS U
GROUP BY U.ID
HAVING ISSUES_QUANTITY > 1;
